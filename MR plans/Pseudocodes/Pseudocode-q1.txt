class Mapper
  method map (lineNo, line) 
    fields = split line by comma
    ID = fields[0];
    Name = fields[1];
    Medal = fields[14];
    
    emit(pair(ID,Name), Medal);  --- this ensures that all records with the same ID and Name will be sent to the same reducer for processing
  
   
class Reducer
   method reduce (pair p, medals [m1, m2, ...])   
   goldCount = 0;
   for all m in medals do 
     if m equals "Gold" 
      goldCount ++;
    
    emit(pair p, goldCount);
